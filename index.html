const express = require('express'); const formidable = require('formidable'); const fs = require('fs'); const app = express(); const port = process.env.PORT || 3000; app.use(express.static('public')); app.post('/upload', (req, res) => { const form = new formidable.IncomingForm(); form.uploadDir = 'uploads'; form.parse(req, (err, fields, files) => { if (err) { res.status(500).send('Internal Server Error'); return; } const videoPath = files.videoInput.path; const videoName = files.videoInput.name; // Move the video file to a permanent location const newPath = `uploads/${videoName}`; fs.rename(videoPath, newPath, (err) => { if (err) { res.status(500).send('Internal Server Error'); return; } res.status(201).send('Video uploaded successfully!'); }); }); }); app.listen(port, () => { console.log(`Server running at http://localhost:${port}`); });
